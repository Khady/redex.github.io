open InfiniteJest

let myTests =
  open InfiniteJest.Test in
  
  describe "Assertions" (fun () -> [
  
    test "completely bananas" (fun () ->
      "bananas" |> Expect.stringContaining "nanas");
      
    test "succ" (fun () ->
      2 |> succ
        |> Expect.toEqual 3);
        
  ])

let _ = 
  run [myTests]
